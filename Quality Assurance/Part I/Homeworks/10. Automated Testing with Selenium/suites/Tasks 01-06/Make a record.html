04. Make a record
open http://demos.kendoui.com/web/listview/editing
click link=Add new record
click css=a.k-button.k-update-button
verifyElementPresent //div[@id='listView']/div/dl/dd/span
verifyElementPresent //div[@id='listView']/div/dl/dd[2]/span[2]
click css=a.k-button.k-edit-button
type //div[@id='listView']/div/dl/dd[1]/input Coffee
fireEvent //div[@id='listView']/div/dl/dd[2]/span/span/input[1] focus
type name=UnitPrice 1
fireEvent name=UnitPrice blur
fireEvent //div[@id='listView']/div/dl/dd[3]/span/span/input[1] focus
type name=UnitsInStock 0
fireEvent name=UnitsInStock blur
click //div[@id='listView']/div/dl/dd[4]/input
storeChecked //div[@id='listView']/div/dl/dd[4]/input discontinued
click css=a.k-button.k-update-button
verifyText //div[@id='listView']/div/dl/dd[1] Coffee
verifyText //div[@id='listView']/div/dl/dd[2] $1.00
verifyText //div[@id='listView']/div/dl/dd[3] 0
verifyText //div[@id='listView']/div/dl/dd[4] ${discontinued}
